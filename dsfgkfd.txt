
CREATE TYPE lectorium_type AS ENUM ('AUDITORIUM', 'LABORATORY', 'COMPUTER_CLASSROOM', 'CLASSROOM', 'COWORKING_SPACE');
CREATE TYPE gender AS ENUM ('MALE', 'FEMALE');

CREATE TABLE Equipment(
    id int primary key generated by default as identity,
    equipment varchar(100) unique
);

CREATE TABLE Organization(
    id int primary key generated by default as identity,
    short_name varchar(10) unique,
    full_name varchar(100) unique
);

CREATE TABLE Discipline(
    id int primary key generated by default as identity,
    name varchar(100) unique
);

CREATE TABLE Location(
    id int primary key generated by default as identity,
    address varchar(100) unique,
    organization int references Organization(id)
);


CREATE TABLE Faculty(
    id int primary key generated by default as identity,
    faculty int,
    organization int references Organization(id)
);

CREATE TABLE Lectorium(
    id int primary key generated by default as identity,
    number varchar(10),
    location int references Location(id),
    size int,
    type lectorium_type,
    equipment int references Equipment(id),
    description varchar(300),
    organization int references Organization(id)
);

CREATE TABLE Tutor(
    id int primary key generated by default as identity,
    firstname varchar(20),
    patronymic varchar(20),
    lastname varchar(20),
    phone_number varchar(12),
    email varchar(100),
    sex gender,
    organization int references Organization(id),
    faculty int references Faculty(id)
);

CREATE TABLE Disciplines(
    id int primary key generated by default as identity,
    discipline int references Discipline(id),
    tutor int references Tutor(id)
);

CREATE TABLE Lecture(
    id int primary key generated by default as identity,
    tutor int references Tutor(id),
    lectorium int references Lectorium(id),
    discipline int references Discipline(id),
    date date,
    start_of_activity time,
    end_of_activity time
);
